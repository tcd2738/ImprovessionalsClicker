Buttons
	*impButton
		name:Make a joke!
		desc:Click The Improvessionals (TM) logo to get more jokes.
		on click:anim icon wobble
		on click:
			if (have customCahDeck)
				if (chance(1%)) yield 1 zinger
			else if (have quoteJar)
				if (chance(1%) and chance(50%)) yield 1 zinger
			end
		end
		on click:
			$totalJokes=1
			$percentOfJps=(0.01*jokes:ps)

			if (have obsession)
				$totalJokes=($totalJokes+($percentOfJps*6))
			else if (have prop)
				$totalJokes=($totalJokes+($percentOfJps*5))
			else if (have animal)
				$totalJokes=($totalJokes+($percentOfJps*4))
			else if (have posture)
				$totalJokes=($totalJokes+($percentOfJps*3))
			else if (have attitude)
				$totalJokes=($totalJokes+($percentOfJps*2))
			else if (have voice)
				$totalJokes=($totalJokes+$percentOfJps)
			end

			if (have weed) 
				$totalJokes=($totalJokes+420)
			end
			if (have onionSubscription) 
				$totalJokes=($totalJokes+20)
			end			
			if (have twitterDotCom) 
				$totalJokes=($totalJokes+5)	
			end
			if (have ucbManual) 
				$totalJokes=($totalJokes+1)
			end

			yield $totalJokes jokes
		end
		icon:https://raw.githubusercontent.com/tcd2738/ImprovessionalsClicker/main/ImprovessionalsLogo.png
		no text
		class:bigButton hasFlares
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red

Shinies
	*quacklemore
		on click:log Woop
		movement:onRight moveLeft fade bounce:0.05
		frequency:60
		frequency variation:30
		icon:https://raw.githubusercontent.com/tcd2738/ImprovessionalsClicker/main/Quacklemore.png
		class:bigButton
		on click:
			$jokeBoost=1
			$zingerBoost=1
			if (have discordServer) $jokeBoost=2
			if (have webbAuditorium) $zingerBoost=2

			if (chance(5%))
				yield 1 toastTheCat
			else if (chance(50%))
				//get at least 7, or between 1 and 3 minutes of our joke production
				$amount=max(7,random(jokes:ps*60*1,jokes:ps*60*3))*$jokeBoost
				toast Quacklemore grants you<//><b>[$amount] jokes</b>.
				yield $amount jokes
			else
				$amount=1*$zingerBoost
				toast Quacklemore grants you<//><b>[$amount] zinger[s?$amount]</b>!
				yield $amount zinger
			end
		end

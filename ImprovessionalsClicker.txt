Shinies
	*quacklemore
		on click:log Woop
		movement:onRight moveLeft fade bounce:0.05
		frequency:60
		frequency variation:30
		icon:https://raw.githubusercontent.com/tcd2738/ImprovessionalsClicker/main/Quacklemore.png
		class:bigButton
		on click:
			$boost=1
			if (chance(100%))
				toast Toast!
			else if (chance(25%))
				//get at least 7, or between 1 and 3 minutes of our comedy point production
				$amount=max(7,random(comedyPoints:ps*60*1,comedyPoints:ps*60*3))*$boost
				toast Quacklemore grants you<//><b>[$amount] comedy points</b>.
				yield $amount comedyPoints
			else
				$amount=1*$boost
				toast Quacklemore grants you<//><b>[$amount] zingers[s?$amount]</b>!
				yield $amount zinger
			end
		end
